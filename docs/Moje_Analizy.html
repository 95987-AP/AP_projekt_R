mtcars_filt <- mtcars %>% filter(hp > 100)

mtcars <- mtcars %>% 
  mutate(kategorie_cyl = case_when(
    cyl == 4 ~ "4 cylindry",
    cyl == 6 ~ "6 cylindrów", 
    cyl == 8 ~ "8 cylindrów"
  ))

mtcars <- mtcars %>% 
  mutate(efficient = ifelse(mpg > median(mpg), "wysoka", "niska"))
ggplot(mtcars, aes(x = hp, y = mpg, color = kategorie_cyl)) +
